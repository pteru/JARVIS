```python
from __future__ import annotations

import json
from dataclasses import dataclass, field
from pathlib import Path


class ConfigError(ValueError):
    pass


@dataclass(frozen=True)
class TagGroup:
    name: str
    interval_ms: int
    tags: tuple[str, ...]


@dataclass(frozen=True)
class CommandConfig:
    max_age_ms: int = 300
    discard_classes: tuple[str, ...] = ("motion",)
    echo_map: dict[str, str] = field(default_factory=dict)


@dataclass(frozen=True)
class Config:
    plc_ip: str
    plc_key: str
    groups: tuple[TagGroup, ...]
    command: CommandConfig


def load_config(path: str | Path) -> Config:
    try:
        raw = json.loads(Path(path).read_text())
    except (OSError, json.JSONDecodeError) as e:
        raise ConfigError(f"cannot read config {path}: {e}") from e

    try:
        plc_ip = raw["plc_ip"]
        plc_key = raw["plc_key"]
        groups = tuple(
            TagGroup(name=g["name"], interval_ms=int(g["interval_ms"]), tags=tuple(g["tags"]))
            for g in raw["groups"]
        )
    except (KeyError, TypeError) as e:
        raise ConfigError(f"missing/invalid field: {e}") from e

    if not groups:
        raise ConfigError("at least one poll group required")
    names = [g.name for g in groups]
    if len(names) != len(set(names)):
        raise ConfigError(f"duplicate group names: {names}")
    for g in groups:
        if g.interval_ms <= 0:
            raise ConfigError(f"group {g.name}: interval_ms must be > 0")
        if not g.tags:
            raise ConfigError(f"group {g.name}: tags must be non-empty")

    cmd = raw.get("command", {})
    command = CommandConfig(
        max_age_ms=int(cmd.get("max_age_ms", 300)),
        discard_classes=tuple(cmd.get("discard_classes", ["motion"])),
        echo_map=dict(cmd.get("echo_map", {})),
    )
    return Config(plc_ip=plc_ip, plc_key=plc_key, groups=groups, command=command)
```

"""In-memory doubles for pylogix.PLC and redis.Redis (only what the client uses)."""
from dataclasses import dataclass


@dataclass
class FakeResponse:
    TagName: str
    Value: object
    Status: str


class FakePLC:
    """Dict-backed tag server. Set read_fail={tag} to simulate per-tag failure."""

    def __init__(self, tags=None, read_fail=None, write_fail=None):
        self.tags = dict(tags or {})
        self.read_fail = set(read_fail or ())
        self.write_fail = set(write_fail or ())
        self.writes = []

    def Read(self, tag_list):
        return [
            FakeResponse(t, None, "Connection failure") if t in self.read_fail
            else FakeResponse(t, self.tags.get(t), "Success")
            for t in tag_list
        ]

    def Write(self, tag, value):
        self.writes.append((tag, value))
        if tag in self.write_fail:
            return FakeResponse(tag, None, "Path segment error")
        self.tags[tag] = value
        return FakeResponse(tag, value, "Success")

    def __enter__(self):
        return self

    def __exit__(self, *exc):
        return False


class FakeRedis:
    def __init__(self):
        self.hashes = {}
        self.stream_adds = []
        self.acks = []

    def hset(self, key, mapping):
        self.hashes.setdefault(key, {}).update(mapping)

    def xadd(self, key, fields):
        self.stream_adds.append((key, fields))
        return b"0-1"

    def xack(self, key, group, entry_id):
        self.acks.append((key, group, entry_id))
```

```python
from __future__ import annotations

import json
import logging
import threading
import time

from .config import TagGroup
from .keys import tags_key

log = logging.getLogger(__name__)


def _now_ms() -> int:
    return time.time_ns() // 1_000_000


def poll_once(plc, redis_client, plc_key: str, group: TagGroup, now_ms: int) -> bool:
    """One batch read -> one hash write. All-or-nothing: on any per-tag failure
    nothing is written, so consumers detect staleness via updated_at (C2 rule 5-a)."""
    responses = plc.Read(list(group.tags))
    failed = [r for r in responses if r.Status != "Success"]
    if failed:
        log.warning("group %s: %d/%d reads failed (first: %s %s)",
                    group.name, len(failed), len(responses), failed[0].TagName, failed[0].Status)
        return False
    mapping = {r.TagName: json.dumps(r.Value) for r in responses}
    mapping["updated_at"] = str(now_ms)
    redis_client.hset(tags_key(plc_key, group.name), mapping=mapping)
    return True


class PollThread(threading.Thread):
    """One thread per group, each with its OWN pylogix instance (pylogix is not thread-safe)."""

    def __init__(self, plc_factory, redis_client, plc_key: str, group: TagGroup,
                 stop_event: threading.Event):
        super().__init__(name=f"poll-{group.name}", daemon=True)
        self._plc_factory = plc_factory
        self._redis = redis_client
        self._plc_key = plc_key
        self._group = group
        self._stop = stop_event

    def run(self):
        with self._plc_factory() as plc:
            while not self._stop.is_set():
                start = time.monotonic()
                poll_once(plc, self._redis, self._plc_key, self._group, _now_ms())
                elapsed_ms = (time.monotonic() - start) * 1000.0
                self._stop.wait(max(0.0, (self._group.interval_ms - elapsed_ms) / 1000.0))
```

```python
from __future__ import annotations

import json
from dataclasses import dataclass

from .config import CommandConfig


class CmdParseError(ValueError):
    pass


@dataclass(frozen=True)
class CmdEntry:
    entry_id: str
    seq: str
    ts_ms: int
    tag: str
    value: object
    cls: str
    group: str


def parse_entry(entry_id: str, fields: dict[str, str]) -> CmdEntry:
    try:
        return CmdEntry(
            entry_id=entry_id,
            seq=fields["seq"],
            ts_ms=int(fields["ts_ms"]),
            tag=fields["tag"],
            value=json.loads(fields["value"]),
            cls=fields["class"],
            group=fields.get("group", ""),
        )
    except (KeyError, ValueError) as e:
        raise CmdParseError(f"bad command entry {entry_id}: {e}") from e


def plan_batch(entries: list[CmdEntry], cmd_cfg: CommandConfig,
               now_ms: int) -> list[tuple[CmdEntry, str]]:
    """C2 rule 6 generalized: stops execute first and supersede pending motion of the
    same group; discard_classes older than max_age_ms are dropped (crash-only backlog)."""
    stops = [e for e in entries if e.cls == "stop"]
    others = [e for e in entries if e.cls != "stop"]
    stopped_groups = {e.group for e in stops}

    plan: list[tuple[CmdEntry, str]] = [(e, "apply") for e in stops]
    for e in others:
        if e.cls == "motion" and e.group in stopped_groups:
            plan.append((e, "superseded"))
        elif e.cls in cmd_cfg.discard_classes and now_ms - e.ts_ms > cmd_cfg.max_age_ms:
            plan.append((e, "discarded_stale"))
        else:
            plan.append((e, "apply"))
    return plan
```

```python
import logging
import threading
import time

from .keys import audit_key, cmd_key

log = logging.getLogger(__name__)

CONSUMER_GROUP = "tag-client"


def _audit(redis_client, plc_key: str, entry: CmdEntry | None, entry_id: str,
           outcome: str, latency_ms: int | None = None, detail: str = "",
           echo: str | None = None) -> None:
    fields = {"outcome": outcome, "detail": detail,
              "seq": entry.seq if entry else "", "entry_id": entry_id}
    if latency_ms is not None:
        fields["latency_ms"] = str(latency_ms)
    if echo is not None:
        fields["echo"] = echo
    redis_client.xadd(audit_key(plc_key), fields)


def execute_batch(plc, redis_client, plc_key: str, cmd_cfg: CommandConfig,
                  raw_entries: list[tuple[str, dict]], now_ms_fn) -> None:
    """Parse -> plan -> write to PLC -> audit -> ack. Every raw entry is ALWAYS
    audited and acked exactly once, including unparseable ones (no poison loop)."""
    entries: list[CmdEntry] = []
    for entry_id, fields in raw_entries:
        try:
            entries.append(parse_entry(entry_id, fields))
        except CmdParseError as e:
            _audit(redis_client, plc_key, None, entry_id, "error", detail=str(e))
            redis_client.xack(cmd_key(plc_key), CONSUMER_GROUP, entry_id)

    for entry, action in plan_batch(entries, cmd_cfg, now_ms_fn()):
        if action == "apply":
            resp = plc.Write(entry.tag, entry.value)
            if resp.Status == "Success":
                echo = None
                if entry.tag in cmd_cfg.echo_map:
                    echo_resp = plc.Read([cmd_cfg.echo_map[entry.tag]])[0]
                    if echo_resp.Status == "Success":
                        echo = json.dumps(echo_resp.Value)
                _audit(redis_client, plc_key, entry, entry.entry_id, "applied",
                       latency_ms=now_ms_fn() - entry.ts_ms, echo=echo)
            else:
                _audit(redis_client, plc_key, entry, entry.entry_id, "error",
                       detail=f"write failed: {resp.Status}")
        else:
            _audit(redis_client, plc_key, entry, entry.entry_id, action)
        redis_client.xack(cmd_key(plc_key), CONSUMER_GROUP, entry.entry_id)


class CommandThread(threading.Thread):
    """Dedicated command thread with its OWN pylogix instance (C2 rule 6.i)."""

    def __init__(self, plc_factory, redis_client, plc_key: str,
                 cmd_cfg: CommandConfig, stop_event: threading.Event):
        super().__init__(name="commands", daemon=True)
        self._plc_factory = plc_factory
        self._redis = redis_client
        self._plc_key = plc_key
        self._cfg = cmd_cfg
        self._stop = stop_event

    def _ensure_group(self):
        import redis as redis_lib
        try:
            self._redis.xgroup_create(cmd_key(self._plc_key), CONSUMER_GROUP,
                                      id="$", mkstream=True)
        except redis_lib.ResponseError as e:
            if "BUSYGROUP" not in str(e):
                raise

    def run(self):
        self._ensure_group()
        now_ms_fn = lambda: time.time_ns() // 1_000_000
        with self._plc_factory() as plc:
            while not self._stop.is_set():
                resp = self._redis.xreadgroup(
                    CONSUMER_GROUP, "worker-1",
                    {cmd_key(self._plc_key): ">"}, count=64, block=1000)
                if not resp:
                    continue
                _stream, raw_entries = resp[0]
                execute_batch(plc, self._redis, self._plc_key, self._cfg,
                              raw_entries, now_ms_fn)

from __future__ import annotations

import logging
import os
import signal
import sys
import threading

import pylogix
import redis

from .command_worker import CommandThread
from .config import Config, load_config
from .poller import PollThread

log = logging.getLogger("eip_tag_client")


def build_threads(cfg: Config, plc_factory, redis_client,
                  stop_event: threading.Event) -> list[threading.Thread]:
    threads: list[threading.Thread] = [
        PollThread(plc_factory, redis_client, cfg.plc_key, g, stop_event)
        for g in cfg.groups
    ]
    threads.append(CommandThread(plc_factory, redis_client, cfg.plc_key,
                                 cfg.command, stop_event))
    return threads


def _crash_hook(args):
    log.critical("thread %s died: %s", args.thread.name if args.thread else "?",
                 args.exc_value, exc_info=(args.exc_type, args.exc_value, args.exc_traceback))
    os._exit(1)  # crash-only: Docker restart heals


def main() -> None:
    logging.basicConfig(level=os.environ.get("LOG_LEVEL", "INFO"),
                        format="%(asctime)s %(levelname)s %(name)s %(threadName)s %(message)s")
    threading.excepthook = _crash_hook

    cfg = load_config(os.environ.get("TAGS_JSON", "/etc/eip/tags.json"))
    redis_client = redis.Redis(
        host=os.environ.get("REDIS_HOST", "redis"),
        port=int(os.environ.get("REDIS_PORT", "6379")),
        password=os.environ.get("REDIS_PASSWORD") or None,
        decode_responses=True,
    )
    redis_client.ping()  # fail fast: sem Redis o serviço não tem função

    def plc_factory():
        plc = pylogix.PLC()
        plc.IPAddress = cfg.plc_ip
        return plc

    stop = threading.Event()
    for sig in (signal.SIGTERM, signal.SIGINT):
        signal.signal(sig, lambda *_: stop.set())

    threads = build_threads(cfg, plc_factory, redis_client, stop)
    for t in threads:
        t.start()
    log.info("eip-tag-client up: %d poll groups + command thread (plc=%s key=%s)",
             len(cfg.groups), cfg.plc_ip, cfg.plc_key)

    while not stop.is_set():
        for t in threads:
            t.join(timeout=0.5)
            if not t.is_alive() and not stop.is_set():
                log.critical("thread %s exited unexpectedly", t.name)
                sys.exit(1)
    for t in threads:
        t.join(timeout=5)


if __name__ == "__main__":
    main()
```

```markdown
# Redis Key Contract — Strokmatic EIP stack

Single IPC contract between the certified adapter, the tag-client and consumer
services (WS). Namespace per `<plc_key>` (convention: the PLC IP).

| Family | Key | Writer | Format |
|---|---|---|---|
| **hot** | `io:in:<plc_key>` / `io:out:<plc_key>` | adapter / plugins | 128-byte binary string (SET / GETRANGE). **Frozen** — certified envelope. |
| **cold** | `tags:<plc_key>:<group>` | tag-client | HASH: one field per tag (JSON-encoded value) + `updated_at` (epoch ms) |
| **cmd** | `cmd:<plc_key>` | consumers (WS) | STREAM — fields below |
| **audit** | `audit:<plc_key>` | tag-client | STREAM — outcome of every consumed command |

## cmd stream fields

| Field | Meaning |
|---|---|
| `seq` | producer-stamped sequence (echoed by PLC when the consumer maps it) |
| `ts_ms` | producer timestamp, epoch ms — basis for age discard |
| `tag` | PLC tag name to write |
| `value` | JSON-encoded scalar |
| `class` | `motion` \| `stop` \| `config` |
| `group` | supersede scope, e.g. axis name |

## Semantics (tag-client, consumer group `tag-client`)

- **Age discard**: entries with `class` in `discard_classes` (default `motion`)
  older than `max_age_ms` (default 300) are discarded — the service is
  crash-only and the stream backlogs during restart; replaying an old jog is
  unrequested motion.
- **Priority stop**: `class=stop` executes before pending entries of the batch
  and supersedes pending `motion` of the same `group`. Stops are never
  age-discarded.
- **Audit**: every consumed entry produces exactly one `audit:<plc_key>` entry
  (`outcome` = `applied` | `discarded_stale` | `superseded` | `error`, plus
  `latency_ms` and optional `echo` when the consumer configured `echo_map`)
  and exactly one XACK — including unparseable entries (no poison loop).
- **Staleness (cold)**: `updated_at` only advances when the whole group batch
  read succeeds; consumers must treat stale hashes as unavailable (C2 rule 5-a
  of the 03007 contract, generalized).

def wait_for(predicate, timeout=5.0):
    deadline = time.monotonic() + timeout

    return False
